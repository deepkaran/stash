=======================================================================================
1. git clone https://github.com/blevesearch/faiss.git
2. cd faiss (make sure you're on latest of bleve branch)
3. Configure
export LDFLAGS="-L/usr/local/opt/llvm/lib/c++ -Wl,-rpath,/usr/local/opt/llvm/lib/c++"
export CPPFLAGS="-I/usr/local/opt/llvm/include"
export CXX=/usr/local/opt/llvm/bin/clang++
export CC=/usr/local/opt/llvm/bin/clang cmake
cmake -B build -DFAISS_ENABLE_GPU=OFF -DFAISS_ENABLE_C_API=ON -DBUILD_SHARED_LIBS=ON -DFAISS_ENABLE_PYTHON=OFF .
4. make -C build
5. sudo make -C build install
6. sudo cp build/c_api/libfaiss_c.dylib /usr/local/lib
